98 static inline std::vector<std::string_view> split(
99 const std::string_view& s,
100 const std::string_view& separator = " ",
101 size_t max_split = SIZE_MAX)
102 {
103 std::vector<std::string_view> result;
104
105 auto separator_end = 0;
106 auto next_separator_start = s.find(separator);
107 while (next_separator_start != std::string_view::npos &&
108 result.size() < max_split)
109 {
110 result.push_back(
111 s.substr(separator_end, next_separator_start - separator_end));
112
113 separator_end = next_separator_start + separator.size();
114 next_separator_start = s.find(separator, separator_end);
115 }
116
117 result.push_back(s.substr(separator_end));
118
119 return result;
120 }

144 static inline std::vector<std::string_view> rsplit(
145 const std::string_view& s,
146 const std::string_view& separator = " ",
147 size_t max_split = SIZE_MAX)
148 {
149 std::vector<std::string_view> result;
150
151 auto prev_separator_start = s.size();
152 auto next_separator_start = s.rfind(separator);
153 while (next_separator_start != std::string_view::npos &&
154 result.size() < max_split)
155 {
156 auto separator_end = next_separator_start + separator.size();
157
158 result.push_back(
159 s.substr(separator_end, prev_separator_start - separator_end));
160
161 prev_separator_start = next_separator_start;
162
163 if (next_separator_start == 0)
164 {
165 break;
166 }
167 else
168 {
169 next_separator_start = s.rfind(separator, prev_separator_start - 1);
170 }
171 }
172
173 result.push_back(s.substr(0, prev_separator_start));
174
175 return result;
176 }

178 /* rsplit_1 wraps rsplit _and reverses the result order_ and allows writing
179 * things like:
180 * auto [host, port] = nonstd::rsplit_1("[1:2:3:4]:8000", ":")
181 */
182 static inline std::tuple<std::string_view, std::string_view> rsplit_1(
183 const std::string_view& s, const std::string_view& separator)
184 {
185 const auto v = rsplit(s, separator, 1);
186 if (v.size() == 1)
187 {
188 // If separator is not present, return {"", s};
189 return std::make_tuple("", v[0]);
190 }
191
192 return std::make_tuple(v[1], v[0]);
